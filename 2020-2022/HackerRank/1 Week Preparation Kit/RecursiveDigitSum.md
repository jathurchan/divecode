def superDigit(n, k):
    # Write your code here
    
    def compute(prev):
        if prev < 10:
            return prev
        
        sum = 0
        temp = prev
        
        while prev > 0:
            sum += (prev % 10)
            prev = prev // 10
        
        return compute(sum)
    
    return compute(int(n)*k)
